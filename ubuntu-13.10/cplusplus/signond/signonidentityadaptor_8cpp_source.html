158 void SignonIdentityAdaptor::remove()
159 {
160  /* Access Control */
161  AccessControlManagerHelper::IdentityOwnership ownership =
162  AccessControlManagerHelper::instance()->isPeerOwnerOfIdentity(
163  parentDBusContext().connection(),
164  parentDBusContext().message(),
165  m_parent->id());
166 
167  if (ownership != AccessControlManagerHelper::IdentityDoesNotHaveOwner) {
168  //Identity has an owner
169  if (ownership == AccessControlManagerHelper::ApplicationIsNotOwner &&
170  !AccessControlManagerHelper::instance()->isPeerKeychainWidget(
171  parentDBusContext().connection(),
172  parentDBusContext().message())) {
173  securityErrorReply(__func__);
174  return;
175  }
176  }
177 
178  m_parent->remove();
179 }
