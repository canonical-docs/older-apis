166 const QDomDocument Provider::domDocument() const
167 {
168  const gchar *data;
169 
170  ag_provider_get_file_contents(m_provider, &data);
171 
172  QDomDocument doc;
173  QString errorStr;
174  int errorLine;
175  int errorColumn;
176  if (!doc.setContent(QByteArray(data), true,
177  &errorStr, &errorLine, &errorColumn))
178  {
179  QString message(ASCII("Parse error reading account provider file "
180  "at line %1, column %2:\n%3"));
181  message.arg(errorLine).arg(errorColumn).arg(errorStr);
182  qWarning() << __PRETTY_FUNCTION__ << message;
183  }
184 
185  return doc;
186 }
