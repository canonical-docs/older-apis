181 QSet<QString> Service::tags() const
182 {
183  if (m_tags)
184  return *m_tags;
185 
186  m_tags = new QSet<QString>;
187  GList *list = ag_service_get_tags(m_service);
188  GList *iter = list;
189  while (iter != NULL) {
190  m_tags->insert(UTF8(reinterpret_cast<const gchar *> (iter->data)));
191  iter = g_list_next(iter);
192  }
193  g_list_free(list);
194  return *m_tags;
195 }

201 const QDomDocument Service::domDocument() const
202 {
203  const gchar *data;
204 
205  ag_service_get_file_contents(m_service, &data, NULL);
206 
207  QDomDocument doc;
208  QString errorStr;
209  int errorLine;
210  int errorColumn;
211  if (!doc.setContent(QByteArray(data), true,
212  &errorStr, &errorLine, &errorColumn))
213  {
214  QString message(ASCII("Parse error reading account service file "
215  "at line %1, column %2:\n%3"));
216  message.arg(errorLine).arg(errorColumn).arg(errorStr);
217  qWarning() << __PRETTY_FUNCTION__ << message;
218  }
219  return doc;
220 }
